\begin{algorithm}
\caption{Roleta de Seleção}
\KwIn{Lista de pares $(objeto, distancia)$}
\KwOut{Objeto selecionado}
\ForEach{$(objeto, distancia)$ em roleta}{
    \If{$distancia = 0$}{
        \Return $objeto$
    }
    Adicione $objeto$ à lista de objetos\;
    Adicione $distancia$ à lista de distancias\;
}

\ForEach{$distancia$ em distancias}{
    Calcule $peso = 1 / distancia$\;
    Adicione $peso$ à lista de pesos\;
}

Calcule $probabilidades = pesos / \sum pesos$\;
Escolha aleatoriamente um objeto com base nas probabilidades\;
\Return objeto escolhido\;

\end{algorithm}


\begin{algorithm}
\caption{Gerar Solução}
\KwOut{Melhor rota gerada}

Inicialize $melhor\_rota\_copia \gets [(inicio, 0)]$\;
$ativos\_copia \gets copia(ativos)$\;
$origem \gets inicio$\;

\While{$ativos\_copia \neq \emptyset$}{
    Inicialize lista $roleta$\;

    \ForEach{$ativo$ em $ativos\_copia$}{
        \If{$ativo \notin distancia\_entre\_ativos[origem]$}{
            $rota, distancia, nos, ramificacao \gets algoritmo(origem, ativo)$\;
            Adicione $rota$ a $caminho\_entre\_ativos$\;
            Atualize $distancia\_entre\_ativos[origem][ativo] \gets distancia$\;
        }
        Adicione $(ativo, distancia)$ à $roleta$\;
    }

    $elemento \gets Roleta(roleta)$\;
    Adicione $(elemento, 0)$ à $melhor\_rota\_copia$\;
    Remova $elemento$ de $ativos\_copia$\;
    $origem \gets elemento$\;
}

\If{$inicio \notin distancia\_entre\_ativos[origem]$}{
    $rota, distancia, nos, ramificacao \gets algoritmo(origem, inicio)$\;
    Adicione $rota$ a $caminho\_entre\_ativos$\;
    Atualize $distancia\_entre\_ativos[origem][inicio] \gets distancia$\;
}
Adicione $(inicio, 0)$ à $melhor\_rota\_copia$\;

Chame $recalcula\_distancias(melhor\_rota\_copia)$\;
\Return $melhor\_rota\_copia$\;

\end{algorithm}
